javascriptCopyfunction extractStudentNames(classId) {
    const imageFile = document.getElementById('studentListImage').files[0];
    
    if (!imageFile) {
        alert('Please select an image first!');
        return;
    }
    
    document.getElementById('extractedNamesContainer').innerHTML = 'Processing image...';
    
    // ONLY USE REAL OCR - NO FAKE NAMES
    Tesseract.recognize(imageFile, 'eng')
        .then(({ data: { text } }) => {
            console.log('OCR Result:', text); // Show what OCR actually found
            
            // Extract actual names from OCR text
            const lines = text.split('\n');
            const realNames = [];
            
            lines.forEach(line => {
                const cleaned = line.trim();
                if (cleaned.length > 2 && /[A-Za-z]/.test(cleaned)) {
                    realNames.push(cleaned);
                }
            });
            
            if (realNames.length === 0) {
                alert('No names found in image. Try a clearer photo.');
                return;
            }
            
            // Show actual extracted names
            showRealExtractedNames(realNames, classId);
        })
        .catch(error => {
            console.error('OCR failed:', error);
            alert('OCR failed. Please try manual entry.');
        });
}

function showRealExtractedNames(names, classId) {
    let html = '<h5>Names found in image:</h5>';
    names.forEach((name, index) => {
        html += `
            <div>
                <input type="text" id="name_${index}" value="${name}" />
                <button onclick="removeName(${index})">Remove</button>
            </div>
        `;
    });
    html += `<button onclick="saveRealNames(${classId}, ${names.length})">Save Names</button>`;
    
    document.getElementById('extractedNamesContainer').innerHTML = html;
}
