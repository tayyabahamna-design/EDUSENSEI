2. FORGET PASSWORD FUNCTIONS
javascriptCopyfunction showForgotPassword() {
    // Hide login form, show forgot password form
    document.querySelector('.login-form').style.display = 'none';
    document.getElementById('forgotPasswordForm').style.display = 'block';
}

function backToLogin() {
    // Show login form, hide forgot password form
    document.querySelector('.login-form').style.display = 'block';
    document.getElementById('forgotPasswordForm').style.display = 'none';
    
    // Reset forgot password form
    resetForgotPasswordForm();
}

function resetForgotPasswordForm() {
    document.getElementById('resetPhone').value = '';
    document.getElementById('verificationCode').value = '';
    document.getElementById('newPassword').value = '';
    document.getElementById('confirmNewPassword').value = '';
    document.getElementById('codeVerification').style.display = 'none';
    document.getElementById('newPasswordSection').style.display = 'none';
}

function sendResetCode() {
    const phone = document.getElementById('resetPhone').value;
    
    if (!phone || phone.length < 11) {
        alert('Please enter a valid phone number!');
        return;
    }
    
    // Check if phone number exists in system
    const existingUsers = JSON.parse(localStorage.getItem('registeredUsers') || '[]');
    const userExists = existingUsers.find(user => user.phone === phone);
    
    if (!userExists) {
        alert('Phone number not found! Please register first.');
        return;
    }
    
    // Generate 4-digit verification code
    const resetCode = Math.floor(1000 + Math.random() * 9000).toString();
    
    // Store reset code temporarily (in real app, send via SMS)
    localStorage.setItem('resetCode', JSON.stringify({
        phone: phone,
        code: resetCode,
        timestamp: Date.now()
    }));
    
    // Show code verification section
    document.getElementById('codeVerification').style.display = 'block';
    
    // Simulate SMS (in real app, integrate SMS service)
    alert(`Reset code sent to ${phone}: ${resetCode}\n(In real app, this will be sent via SMS)`);
}

function verifyResetCode() {
    const enteredCode = document.getElementById('verificationCode').value;
    const storedResetData = JSON.parse(localStorage.getItem('resetCode') || '{}');
    
    if (!enteredCode || enteredCode.length !== 4) {
        alert('Please enter the 4-digit code!');
        return;
    }
    
    // Check if code matches and is not expired (valid for 10 minutes)
    const currentTime = Date.now();
    const codeAge = currentTime - storedResetData.timestamp;
    const tenMinutes = 10 * 60 * 1000;
    
    if (codeAge > tenMinutes) {
        alert('Reset code expired! Please request a new code.');
        localStorage.removeItem('resetCode');
        return;
    }
    
    if (enteredCode === storedResetData.code) {
        // Code verified, show new password section
        document.getElementById('newPasswordSection').style.display = 'block';
        alert('‚úÖ Code verified! Enter your new password.');
    } else {
        alert('‚ùå Invalid code! Please try again.');
    }
}

function resetPassword() {
    const newPassword = document.getElementById('newPassword').value;
    const confirmPassword = document.getElementById('confirmNewPassword').value;
    const resetData = JSON.parse(localStorage.getItem('resetCode') || '{}');
    
    // Validation
    if (!newPassword || newPassword.length < 6) {
        alert('Password must be at least 6 characters long!');
        return;
    }
    
    if (newPassword !== confirmPassword) {
        alert('Passwords do not match!');
        return;
    }
    
    // Update user's password
    const existingUsers = JSON.parse(localStorage.getItem('registeredUsers') || '[]');
    const userIndex = existingUsers.findIndex(user => user.phone === resetData.phone);
    
    if (userIndex !== -1) {
        existingUsers[userIndex].password = newPassword;
        localStorage.setItem('registeredUsers', JSON.stringify(existingUsers));
        
        // Clear reset code
        localStorage.removeItem('resetCode');
        
        // Show success and return to login
        alert('‚úÖ Password reset successfully! Please login with your new password.');
        backToLogin();
    } else {
        alert('‚ùå Error updating password. Please try again.');
    }
}

javascriptCopy// Add rate limiting for reset attempts
function checkResetAttempts(phone) {
    const attempts = JSON.parse(localStorage.getItem('resetAttempts') || '{}');
    const userAttempts = attempts[phone] || { count: 0, lastAttempt: 0 };
    
    const now = Date.now();
    const oneHour = 60 * 60 * 1000;
    
    // Reset attempts after 1 hour
    if (now - userAttempts.lastAttempt > oneHour) {
        userAttempts.count = 0;
    }
    
    // Limit to 3 attempts per hour
    if (userAttempts.count >= 3) {
        alert('Too many reset attempts. Please try after 1 hour.');
        return false;
    }
    
    // Increment attempts
    userAttempts.count++;
    userAttempts.lastAttempt = now;
    attempts[phone] = userAttempts;
    localStorage.setItem('resetAttempts', JSON.stringify(attempts));
    
    return true;
}
