2. REMOVE FAKE NAMES - USE REAL OCR
javascriptCopyfunction extractStudentNames(classId) {
    const files = document.getElementById('studentImages').files;
    
    if (files.length === 0) {
        alert('Please select images first!');
        return;
    }
    
    // Show loading
    document.getElementById('extractedNamesContainer').innerHTML = `
        <p>Processing images... Please wait.</p>
    `;
    
    // Use Tesseract OCR - REAL extraction
    Tesseract.recognize(files[0], 'eng')
    .then(result => {
        const text = result.data.text;
        const lines = text.split('\n');
        const names = lines.filter(line => 
            line.trim().length > 3 && 
            /^[A-Za-z\s]+$/.test(line.trim())
        );
        
        showExtractedNames(names, classId);
    })
    .catch(error => {
        alert('OCR failed. Please try manual entry.');
        console.log('OCR Error:', error);
    });
}

javascriptCopyfunction saveExtractedStudents(classId) {
    const nameInputs = document.querySelectorAll('.name-input');
    const students = [];
    
    nameInputs.forEach(input => {
        if (input.value.trim()) {
            students.push({
                id: Date.now() + Math.random(),
                name: input.value.trim(),
                fatherName: '',
                motherName: '',
                contact: '',
                address: ''
            });
        }
    });
    
    // FORCE SAVE TO LOCALSTORAGE
    let classes = JSON.parse(localStorage.getItem('teacherClasses') || '[]');
    const classIndex = classes.findIndex(c => c.id == classId);
    
    if (classIndex !== -1) {
        classes[classIndex].students = students;
        localStorage.setItem('teacherClasses', JSON.stringify(classes));
        alert('Students saved successfully!');
        openClass(classId);
    }
}
