javascriptCopyfunction handleFileUpload(file) {
    if (file.type.startsWith('image/')) {
        // Image processing
        const reader = new FileReader();
        reader.onload = function(e) {
            sendToUDost(`Teacher uploaded an image. Please analyze and help.`, 'image', e.target.result);
        };
        reader.readAsDataURL(file);
    }
}
