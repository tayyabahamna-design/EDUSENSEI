// Check if user already logged in on page load
window.onload = function() {
    const savedUser = localStorage.getItem('ustaadDostUser');
    
    if (savedUser) {
        const userData = JSON.parse(savedUser);
        const loginTime = userData.loginTime;
        const currentTime = new Date().getTime();
        
        // Keep session active for 7 days
        const sevenDays = 7 * 24 * 60 * 60 * 1000;
        
        if (currentTime - loginTime < sevenDays) {
            // User still logged in, go to dashboard
            showDashboard();
        } else {
            // Session expired, show login
            showLoginPage();
            localStorage.removeItem('ustaadDostUser');
        }
    } else {
        // No saved login, show login page
        showLoginPage();
    }
};
