function verifyResetCode() {

    const enteredCode = document.getElementById('verificationCode').value;

    if (!enteredCode || enteredCode.length !== 4) {

        alert('Please enter the 4-digit code!');

        return;

    }

    

    // Check if code matches and is not expired (valid for 10 minutes)

    const currentTime = Date.now();

    const codeAge = currentTime - storedResetData.timestamp;

    const tenMinutes = 10 * 60 * 1000;

    

    if (codeAge > tenMinutes) {

        alert('Reset code expired! Please request a new code.');

    const confirmPassword = document.getElementById('confirmNewPassword').value;

        return;

        alert('Passwords do not match!');

        return;

    const userIndex = existingUsers.findIndex(user => user.phone === resetData.phone);

    

    if (userIndex !== -1) {

        existingUsers[userIndex].password = newPassword;

        // Clear reset code

        localStorage.removeItem('resetCode');

        alert('‚úÖ Password reset successfully! Please login with your new password.');

function checkResetAttempts(phone) {

    const attempts = JSON.parse(localStorage.getItem('resetAttempts') || '{}');

    const userAttempts = attempts[phone] || { count: 0, lastAttempt: 0 };

    

    const now = Date.now();

    const oneHour = 60 * 60 * 1000;

    

    // Reset attempts after 1 hour

    if (now - userAttempts.lastAttempt > oneHour) {

        userAttempts.count = 0;

    }

    

    // Limit to 3 attempts per hour

    if (userAttempts.count >= 3) {

        alert('Too many reset attempts. Please try after 1 hour.');

        return false;

    }

    

    // Increment attempts

    userAttempts.count++;

    userAttempts.lastAttempt = now;

    attempts[phone] = userAttempts;

    localStorage.setItem('resetAttempts', JSON.stringify(attempts));

    

    return true;

}

function copyToYear() {

    const weeklySchedule = {};

    const days = ['Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday'];

    

    days.forEach(day => {

        const daySchedule = {

            periods: [],

            breaks: []

        };

        

        // Get all periods for this day

        const periodRows = document.querySelectorAll(`#${day}_periods .period-row`);

        periodRows.forEach((row, index) => {

            const periodNum = index + 1;

            const startTime = document.getElementById(`${day}_period${periodNum}_start`)?.value;

            const endTime = document.getElementById(`${day}_period${periodNum}_end`)?.value;

            const subject = document.getElementById(`${day}_period${periodNum}_subject`)?.value;

            

            if (startTime && endTime) {

                daySchedule.periods.push({

                    number: periodNum,

                    startTime: startTime,

                    endTime: endTime,

                    subject: subject || 'Free Period'

                });

            }

        });

        

        // Get break times

        const breakStart = document.getElementById(`${day}_break_start`)?.value;

        const breakEnd = document.getElementById(`${day}_break_end`)?.value;

        const lunchStart = document.getElementById(`${day}_lunch_start`)?.value;

        const lunchEnd = document.getElementById(`${day}_lunch_end`)?.value;

        

        if (breakStart && breakEnd) {

            daySchedule.breaks.push({ type: 'Break', start: breakStart, end: breakEnd });

        }

        if (lunchStart && lunchEnd) {

            daySchedule.breaks.push({ type: 'Lunch', start: lunchStart, end: lunchEnd });

        }

        

        weeklySchedule[day] = daySchedule;

    });

    

    // Save complete schedule

    localStorage.setItem('yearlyPlan', JSON.stringify({

        schedule: weeklySchedule,

        createdDate: new Date().toISOString()

    }));

    

    alert(`‚úÖ Complete schedule with ${Object.keys(weeklySchedule).length} days copied to entire year!`);

    showYearlyCalendar();

function saveAndCopyToYear() {

    const weeklyTemplate = {};

    const days = ['Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday'];

    

    // Save weekly template

    days.forEach(day => {

        weeklyTemplate[day] = {};

        for (let period = 1; period <= 8; period++) {

            weeklyTemplate[day][`period${period}`] = {

                subject: document.getElementById(`${day}_period${period}_subject`).value,

                grade: document.getElementById(`${day}_period${period}_grade`).value,

                startTime: document.getElementById(`${day}_period${period}_start`).value,

                endTime: document.getElementById(`${day}_period${period}_end`).value

            };

        }

        weeklyTemplate[day].break = {

            startTime: document.getElementById(`${day}_break_start`).value,

            endTime: document.getElementById(`${day}_break_end`).value

        };

    });

    

    // Copy to entire year (365 days)

    const yearlyPlan = {};

    for (let month = 1; month <= 12; month++) {

        for (let date = 1; date <= 31; date++) {

            const dateKey = `${month}-${date}`;

            const dayOfWeek = new Date(2024, month-1, date).getDay(); // 0=Sunday, 1=Monday

            

            if (dayOfWeek >= 1 && dayOfWeek <= 5) { // Monday to Friday

                const dayName = days[dayOfWeek - 1];

                yearlyPlan[dateKey] = JSON.parse(JSON.stringify(weeklyTemplate[dayName]));

            }

        }

    }

    

    localStorage.setItem('yearlyPlan', JSON.stringify(yearlyPlan));

    localStorage.setItem('weeklyTemplate', JSON.stringify(weeklyTemplate));

    

    showMonthlyView();

}

    const yearlyPlan = {};

    `;

    

    try {

        let allExtractedText = '';

        

        // Process first image

        if (image1) {

            document.getElementById('ocrStatusText').textContent = 'Processing Image 1...';

            const result1 = await Tesseract.recognize(image1, 'eng', {

                logger: m => {

                    if (m.status === 'recognizing text') {

                        const progress = Math.round(m.progress * 50); // 50% for first image

                        document.getElementById('ocrProgress').style.width = progress + '%';

                    }

                }

            });

            allExtractedText += result1.data.text + '\n';

        }

        

        // Process second image if exists

        if (image2) {

            document.getElementById('ocrStatusText').textContent = 'Processing Image 2...';

            const result2 = await Tesseract.recognize(image2, 'eng', {

                logger: m => {

                    if (m.status === 'recognizing text') {

                        const progress = 50 + Math.round(m.progress * 50); // 50% + second half

                        document.getElementById('ocrProgress').style.width = progress + '%';

                    }

                }

            });

            allExtractedText += result2.data.text;

        }

        

        // Extract names from text

        const extractedNames = extractNamesFromText(allExtractedText);

        

        if (extractedNames.length === 0) {

            alert('‚ùå No names found in the images. Please try with clearer images or use manual entry.');

            return;

        }

        

        displayExtractedNames(extractedNames);

        

    } catch (error) {

        console.error('OCR Error:', error);

        document.getElementById('ocrStatus').innerHTML = `

            <div class="error">

                ‚ùå OCR failed. Please try again with clearer images or use manual entry.

                <br><small>Error: ${error.message}</small>

            </div>

        `;

    }

}



function extractNamesFromText(text) {

    console.log('Raw OCR Text:', text); // Debug log

    

    const lines = text.split('\n')

        .map(line => line.trim())

        .filter(line => line.length > 2); // Filter out very short lines

    

    const namePattern = /^[A-Za-z\s]{3,}$/; // Names with only letters and spaces, at least 3 chars

    const potentialNames = [];

    

    lines.forEach(line => {

        // Clean the line

        const cleaned = line.replace(/[^A-Za-z\s]/g, '').trim();

        

        if (cleaned && namePattern.test(cleaned)) {

            // Split if multiple names on one line

            const words = cleaned.split(/\s+/);

            

            if (words.length >= 2 && words.length <= 4) { // Typical name length

                potentialNames.push(cleaned);

            } else if (words.length > 4) {

                // Try to split into multiple names

                for (let i = 0; i < words.length - 1; i += 2) {

                    if (words[i] && words[i + 1]) {

                        potentialNames.push(`${words[i]} ${words[i + 1]}`);

                    }

                }

            }

        }

    });

    

    // Remove duplicates and return

    return [...new Set(potentialNames)].slice(0, 50); // Limit to 50 names max

}

    document.getElementById('extractedNamesList').innerHTML = extractedHTML;

    window.extractedCount = names.length;

function extractStudentNames(classId) {

    const imageFile = document.getElementById('studentListImage').files[0];

    

    if (!imageFile) {

        alert('Please select an image first!');

        return;

    }

    

    document.getElementById('extractedNamesContainer').innerHTML = 'Reading handwritten names...';

    

    // HANDWRITING OCR SETTINGS

    Tesseract.recognize(imageFile, 'eng', {

        tessedit_char_whitelist: 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz ',

        tessedit_pageseg_mode: 6,

        preserve_interword_spaces: '1'

    })

    .then(({ data: { text } }) => {

        console.log('Raw OCR:', text);

        

        // FILTER OUT GARBAGE

        const lines = text.split('\n');

        const cleanNames = [];

        

        lines.forEach(line => {

            let cleaned = line.trim();

            // Remove symbols and numbers

            cleaned = cleaned.replace(/[^A-Za-z\s]/g, '');

            // Remove single letters

            cleaned = cleaned.replace(/\b[A-Za-z]\b/g, '');

            // Clean extra spaces

            cleaned = cleaned.replace(/\s+/g, ' ').trim();

            

            // Only keep if looks like a name (2+ words, 5+ chars total)

            if (cleaned.length >= 5 && cleaned.split(' ').length >= 2) {

                cleanNames.push(cleaned);

            }

        });

        

        if (cleanNames.length === 0) {

            alert('‚ùå Handwriting unclear. Please use MANUAL ENTRY instead.');

            manualStudentAdd(classId);

            return;

        }

        

        showExtractedNames(cleanNames, classId);

    })

    .catch(error => {

        alert('OCR failed for handwriting. Use MANUAL ENTRY.');

        manualStudentAdd(classId);

    });

}
